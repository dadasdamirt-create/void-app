const tg = window.Telegram.WebApp; tg.expand(); const balanceDisplay = document.getElementById('balance'); const globalDisplay = document.getElementById('global-matter'); const sphere = document.getElementById('void-sphere'); const costClickEl = document.getElementById('cost-click'); // Убедись, что такие ID есть в HTML const costAutoEl = document.getElementById('cost-auto'); const userId = tg.initDataUnsafe?.user?.id || 'guest'; const userName = tg.initDataUnsafe?.user?.first_name || 'Guest'; const referrerId = tg.initDataUnsafe?.start_param || null; let currentBalance = 0; let autoPower = 0; async function apiRequest(action = 'load') { try { const res = await fetch('/api/click', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId, userName, action, referrerId }) }); const data = await res.json(); currentBalance = data.balance; autoPower = data.autoPower || 0; balanceDisplay.innerText = currentBalance.toFixed(4); globalDisplay.innerText = data.globalMatter.toFixed(4); // Обновляем текст на кнопках апгрейда if (costClickEl) costClickEl.innerText = data.nextClickCost.toFixed(4); if (costAutoEl) costAutoEl.innerText = data.nextAutoCost.toFixed(4); } catch (e) { console.error(e); } } // Визуальное обновление каждую секунду setInterval(() => { if (autoPower > 0) { currentBalance += autoPower; balanceDisplay.innerText = currentBalance.toFixed(4); } }, 1000); // Синхронизация раз в 10 сек setInterval(() => apiRequest('load'), 10000); // Функция покупки async function buyUpgrade(type) { const res = await fetch('/api/upgrade', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId, type }) }); const data = await res.json(); if (data.success) { tg.showAlert('Улучшено!'); apiRequest('load'); } else { tg.showAlert(data.message); } } sphere.onclick = (e) => { // Визуальный эффект const flyText = document.createElement('div'); flyText.innerText = '+'; flyText.style.position = 'fixed'; flyText.style.left = e.clientX + 'px'; flyText.style.top = e.clientY + 'px'; flyText.style.animation = 'flyUp 1s forwards'; document.body.appendChild(flyText); setTimeout(() => flyText.remove(), 1000); if (tg.HapticFeedback) tg.HapticFeedback.impactOccurred('light'); // Временно прибавляем визуально (базовый клик, сервер пришлет точный) currentBalance += 0.0001; balanceDisplay.innerText = currentBalance.toFixed(4); apiRequest('click'); }; apiRequest('load');
